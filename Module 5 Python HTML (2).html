variables = ["age", "bmi", "past_consultations", "num_of_steps", 
             "Hospital_expenditure", "NUmber_of_past_hospitalizations", "Anual_Salary"]

fig, axes = plt.subplots(3, 3, figsize=(15, 12))
axes = axes.flatten()

for i, var in enumerate(variables):
    clean = df[[var, 'charges']].dropna()
    axes[i].scatter(clean[var], clean['charges'], color='blue', alpha=0.3)
    m, b = np.polyfit(clean[var], clean['charges'], 1)
    axes[i].plot(clean[var], m * clean[var] + b, color='red')
    axes[i].set_xlabel(var)
    axes[i].set_ylabel('Charges')
    axes[i].set_title(f'{var} vs Charges')

plt.suptitle('Scatter Plots with Regression Line', fontsize=16)
plt.tight_layout()
plt.show()
